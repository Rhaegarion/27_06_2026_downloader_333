def _all_gists_from_records(records: list) -> str:
    """Build concatenated gists text for LLM name extraction."""
    return "\n".join(
        f"[Event {r.event_id} | {r.date} | "
        f"{r.calling_no or ''} -> {r.called_no or ''}]: {r.msg_gist}"
        for r in records if r.msg_gist
    )
PYEOF
